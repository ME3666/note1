#include<bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    sort(s.begin(),s.end());//保证初始顺序
    set<string>res;
    do
    {
        res.insert(s);
    }while(next_permutation(s.begin(),s.end()));
    cout<<res.size()<<endl;
    for(const string &p:res )//为什么用引用？//为了提高效率，避免复制字符串
    {
        cout<<p<<endl;

    }

    return 0;

}

test2(位运算，二进制)

  • m < (1 << n)
    • 左移运算符，表示1的二进制向左移动n位，如果n=5，就是100000(二进制) 25(十进制)
    • 那么m小于它，表示m的二进制小于100000，也就是说，m的范围是00000到11111(二进制)，0-31(十进制)
  • m & (1 << j)
    • 二进制从右向左，第0位，第1位，第2位...
    • & 是按位与运算符，用于检查 m 的第 j 位是否为 1。
    • 只有当m和(1<<j)在同一个位置都使1的时候才是1
    • 比如，0010 & 1110的结果就是0010
    • 如果 m 的第 j 位是 1，则 mask & (1 << i) 的结果为1，条件为真。
    • 如果 m 的第 i 位是 0，则 mask & (1 << i) 的结果为0，条件为假。
  • min(a,b)返回a,b最小值
  • long long minres = LLONG_MAX;使用循环来更新最小值时，不要忘记，在开头把初始值规定为最大值
有 n 个苹果，其重量已知。你的任务是将这些苹果分成两组，使得这两组苹果的重量之差最小。

输入
第一行输入一个整数 n：苹果的数量。

接下来一行输入 n 个整数 p1,p2,p3...每个苹果的重量。

输出
输出一个整数：两组苹果重量之差的最小值。

约束
1≤n≤20
1≤pi≤10^9
​

输入样例
5
3 2 7 4 1
输出样例
1
  • 题解
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n; // 输入苹果数量

    vector<int> p(n); // 存储每个苹果的重量
    for (int i = 0; i < n; i++) {
        cin >> p[i];
    }

    long long minres = LLONG_MAX; // 初始化最小差值为最大值

    // 枚举所有可能的分组方式
    for (int m = 0; m < (1 << n); m++) {
        long long num1 = 0, num2 = 0; // 第一组和第二组的重量

        // 根据 m 的二进制位决定每个苹果的分组
        for (int j = 0; j < n; j++) {
            if (m & (1 << j)) {
                num1 += p[j]; // 放入第一组
            } else {
                num2 += p[j]; // 放入第二组
            }
        }

        // 更新最小差值
        minres = min(minres, abs(num1 - num2));
    }

    cout << minres << endl; // 输出最小差值

    return 0;
}

test3(递归，回溯)

题面
你的任务是在国际象棋棋盘上放置八个皇后，使得任意两个皇后之间都没有攻击。作为额外的挑战，每个方格要么是空闲的，要么是预留的，你只能将皇后放置在空闲的方格上。然而，预留的方格并不会阻止皇后之间的攻击。

有多少种可能的方法可以放置这八个皇后？

输入
输入有八行，每行有八个字符。每个方格要么是空闲的（.），要么是预留的（*）。

输出一个整数：放置皇后的可能方法数。

#include <iostream>
#include <vector>
using namespace std;

vector<string> board(8); // 棋盘
int count = 0; // 合法的放置方式数

// 检查当前位置是否安全
bool is_safe(int row, int col, vector<int> &queens) {
    for (int i = 0; i < row; i++) {
        if (queens[i] == col || abs(queens[i] - col) == abs(i - row)) {
            return false; // 冲突
        }
    }
    return true; // 安全
}

// 递归放置皇后
void solve(int row, vector<int> &queens) {
    if (row == 8) {
        count++; // 找到一种合法的放置方式
        return;
    }

    for (int col = 0; col < 8; col++) {
        if (board[row][col] == '.' && is_safe(row, col, queens)) {
            queens[row] = col; // 放置皇后
            solve(row + 1, queens); // 递归放置下一行的皇后
            queens[row] = -1; // 回溯，移除皇后
        }
    }
}

int main() {
    // 读取棋盘
    for (int i = 0; i < 8; i++) {
        cin >> board[i];
    }

    vector<int> queens(8, -1); // 记录每行皇后所在的列
    solve(0, queens); // 从第 0 行开始放置皇后

    cout << count << endl; // 输出合法的放置方式数

    return 0;
}
